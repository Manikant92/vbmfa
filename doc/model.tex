  P(\Xx)=N(\Xx|0,I_{Q})
\end{align}
\begin{align}
  P(\Xz|\Xp)=Mult(\Xz|\Xp)

\begin{align}
  P(\Xp|\alpha)=Dir(\Xp|\alpha)

\begin{align}
  P(\Lambda^k)=\prod_{q=1}^Q N({\Xl}_{:q}|0,I_P\frac{1}{\nu_q^{s}})
\end{align}
\begin{align}
  P(\Xn)=\prod_{q=1}^Q\Gamma(\nu_q^s|a,b)
\end{align}
